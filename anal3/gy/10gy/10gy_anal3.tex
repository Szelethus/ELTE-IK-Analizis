	\subsection{Folytatás}
	\begin{task}
		\[ f(x,y)=\begin{cases}
			(x^2+y^2)\cdot\sin\frac{1}{x^2+y^2},\quad x^2+y^2\not=0\\
			0,\quad x^2+y^2=0
		\end{cases} \]
		Feladat:
		\begin{enumerate}
			\item $\delta_1f, \delta_2f=?$
			\item $\delta_1f, \delta_2f\notin C\{(0,0)\}$
			\item $f\in D\{(0,0)\}$
		\end{enumerate}
		Megoldás: 
		\[ \forall(x,y)\in\R^2\setminus\{(0,0)\}\Rightarrow \]
		\[ \delta_1f(x,y)=2x\cdot\sin\frac{1}{x^2+y^2}+(x^2+y^2)\cdot\cos\frac{1}{x^2+y^2}\cdot(-1)(x^2+y^2)\cdot2x=2x\cdot\sin\frac{1}{x^2+y^2}-\frac{2x}{x^2+y^2}\cdot\frac{1}{x^2+y^2} \]
		\[ \delta_1f(0,0)=\lim_{x\to0}\frac{f(x,0)-f(0,0)}{x-0}=\lim_{x\to0}\frac{x^2\cdot\sin\frac{1}{x^2}}{x}=\lim_{x\to0}(x\cdot\sin\frac{1}{x^2})=0 \]
		Összefoglalva:
		\[ \delta_1f(x,y)=\begin{cases}
			2x\sin\frac{1}{x^2+y^2}-\frac{2x}{x^2+y^2}\cos\frac{1}{x^2+y^2}\quad (x,y)\not=(0,0)\\
			0;\quad (x,y)=(0,0)
		\end{cases} \]
		Hasonlóan
		\[ \delta_2f(x,y)=\begin{cases}
			?\quad \text{HF.}\\
			0,\quad (x,y)=(0,0)
		\end{cases} \]
		\[ \delta_2f(0,0)=\lim_{y\to0}\frac{f(0,y)-f(0,0)}{y-0}=\lim_{y\to0}(y\cdot\sin\frac{1}{x^2})=0 \]
		Ha $f\in D\{(0,0)\}\quad \Rightarrow\quad f'(0,0)=\big(\delta_1f(0,0),\delta_2f(0,0)\big) =(0,0)=\grad f(0,0)$.
		(iii)
		\[ \lim_{(x,y)\to(0,0)}\frac{|f(x,y)-f(0,0)-\delta_1f(0,0)x-\delta_2f(0,0)y}{\norm{(x,y)}_{\R^2} }=\lim_{(x,y)\to(0,0)}\frac{\left|(x^2+y^2)\sin\frac{1}{x^2+y^2}\right|}{\norm{(x,y)}_{2}}= \]
		Itt célszerű $x^2+y^2$ miat kettes normát váalsztani.
		\[ \lim_{(x,y)\to(0,0)}\sqrt{x^2+y^2}\cdot\left|\sin\frac{1}{x^2+y^2}\right|=0 \]
		$\Rightarrow\quad f\in D\{(0,0)\}$ és $f'(0,0)=(0,0)$.
		
		(ii). 
		\[ \delta_1f(0,0)=0,\quad \text{elég azonban belátni hogy}\quad \lim_{(x,y)\to(0,0)}\delta_1f(x,y)\not=0 \]
		Lehet érezni, hogy a $\delta_1f(x,y)$ függvény $\frac{2x}{x^2+y^2}$ tagjával lesz baj. Ha letakarjuk $y$-t, akkor a tört $\pm\infty$-hez tart. Így tekintsük et
		\[ y=0\quad \text{mentén}\quad \Rightarrow\quad \delta_1f(x,0)=2x\sin\frac{1}{x^2}-\frac{2}{x}\cdot\cos\frac{1}{x^2} \]
		Átviteli elv: Válasszuk az $(x_n)$ sorozatot úgy, hogy $\cos\frac{1}{x^2_n}=1$ legyen:
		\[ \frac{1}{x^2_n}=2n\pi,\quad n\in\N\quad \Rightarrow\quad x_n:=\frac{1}{\sqrt{2n\pi}}\to0\quad \text{ha}\quad (n\to\infty) \]
		Ekkor:
		\[ \delta_1f(x_n,0)=\frac{2}{\sqrt{2n\pi}}\sin(2n\pi)-2\sqrt{2n\pi}\cos(2n\pi)=-2\sqrt{qn\pi}\to-\infty\quad \text{ha}\quad n\to\infty \]
		$\Rightarrow \delta_1f\notin C\{(0,0)\}$
	\end{task}

	\begin{task}
		\[ f(x,y):=\begin{cases}
			xy\cdot\frac{x^2-y^2}{x^2+y^2},\quad (x,y)\in \R^2\setminus\{(0,0)\}\\
			0,\quad (x,y)=(0,0)
		\end{cases} \]
		Lássuk be, hogy $\delta_{12}f(0,0)\not=\delta_{21}f(0,0)$
		
		\textit{Megoldás:}
		\[ \delta_{12}f(0,0)=\delta_1(\delta_2f)(0,0)=\lim_{x\to0}\frac{\delta_2f(x,0)-\delta_2f(0,0)}{x-0}=\lim_{x\to0}\frac{1}{x}\left[\lim_{y\to0}\frac{f(x,y)-f(x,0)}{y-0}-\lim_{\to0}\frac{f(0,y)-f(0,0)}{y-0}\right]= \]
		\[= \lim_{x\to0}\frac{1}{x}\left[\lim_{y\to0}\frac{xy\cdot\frac{x^2-y^2}{x^2+y^2}-0}{y}-\lim_{y\to0}\frac{0-0}{y}\right]=\lim_{x\to0}\frac{1}{x}\cdot\lim_{y\to0}\left(x\frac{x^2-y^2}{x^2+y^2}\right)=\lim_{x\to0}\left(\frac{1}{x}\cdot x\cdot\frac{x^2}{x^2}\right)=\lim_{x\to0}(1)=1 \]
	\end{task}

	\subsection{Taylor-formula}
	\begin{revision}
		\[ 1\leq n\in\N;\quad f\in\R^n \to\R;\quad a\in\Int D_f;\quad s\in\N ;\quad \exists k(a):\quad \forall x\in k(a):\quad f\in D^{s+1}\{x\}\quad \Rightarrow\]
		\[\quad \forall h\in \R^n\setminus\{0\}\quad \exists\nu(0,1) \]
		\[ f(a+h)=f(a)+\sum_{k=1}^s \sum_{ \substack{i\in\N^n\\|i|=k} }\frac{\delta^if(a)}{i!}h^i+\sum_{i\in\N^n}\frac{\delta^if(a+\nu h)}{i!}h^i \]
	\end{revision}
	\begin{example}
		\[ f(x,y)=x^2y+xy^2-2xy\quad ((x,y)\in\R^2) \]
		Adjuk meg azon $a_{nk}$ együtthatókat, melekkel felírható:
		\[ f(x,y)=\sum_{n,k=0}^{+\infty}a_{nk}\cdot(x-1)^n(y+1)^k \]
		(azaz $(x-1)^n(y+1)^n$ lineáris kombinációjaként)
		%TODO 01 Melyik elsőfokú függvény közelíti a legjobban? T_0f(x)=f(a)
		%TODO másodfokú T_1f(x)=f(a)+f'(a)(x-a)
		%TODO másodfokú: T_2f(x)=f(a)+f'(a)(x-a)+\frac{f''(a)}{2!}(x-a)^2
		$a:=(1,-1);\quad a+h:=(x,y)\quad \Rightarrow\quad h=(x,y)-(1,-1)=(x-1,y+1)$
		\[ |i|=i_1+i_2+\ldots+i_n\quad i!:=i_1!i_2!\cdot\ldots\cdot i_n!;\quad \delta^if(a):=\delta_{\underbrace{1\cdots1}_{i_1}}\delta_{\underbrace{2\cdots2}_{i_2}}\cdots\delta_{\underbrace{n\cdots n}_{i_n}}f(a);\quad h^i=h_1^{i_1}\cdot h_2^{i_2}\dots h_n^{i_n}\quad h\in\R^n \]
		Folytatván:
		\[ f(1,-1)=2\quad (k=0) \]
		\[ k=1\quad |i|=2 \]
		\[ i=(1;0)\to\frac{\delta_1f(1,-1)}{1!\cdot0!}(x-1)^1(y+1)^0 \]
		\[ i=(0,1)\to\frac{\delta_2f(1,-1)}{0!1!}(x-1)^0(y+1)^1 \]
		Tehát au 1. fokú tagok:
		\[ \delta_1f(1,-1)(x-1)+\delta_2f(1,-1)(y+1); \]
		\begin{align*}
			k=2&\quad \Rightarrow\quad i\in\N^2,\quad |i|=2\quad \Rightarrow\quad\\
			i=(2,0)&\quad \to\quad \frac{\delta_{11}f(1,-1)}{2!0!}(x-1)^2(y+1)^0\\
			i=(1,1)&\quad \to\quad \frac{\delta_{12}f(1,-1)}{1!1!}(x-1)^1(y+1)^1\\
			i=(0,2)&\quad \to\quad \frac{\delta_{22}f(1,-1)}{0!2!}(x-1)^0(y+1)^2
		\end{align*}
		Befejezésül:
		\[ \frac{1}{2}\delta_{11}f(1,-1)(x-1)^2+\delta_{12}f(1,-1)(x-1)(y+1)+\frac{1}{2}\delta_{22}f(1,-1)(y+1)^2 \]
		\begin{align*}
			k=3&\quad \Rightarrow\quad i\in\N^2,\quad |i|=3\quad \Rightarrow\quad\\
			i=(3,0)&\quad \to\quad \frac{\delta_{111}f(3,-1)}{2!0!}(x-1)^3(y+1)^0\\
			i=(2,1)&\quad \to\quad \frac{\delta_{112}f(1,-1)}{2!1!}(x-1)^2(y+1)^1\\
			i=(1,2)&\quad \to\quad \frac{\delta_{122}f(1,-1)}{0!2!}(x-1)^1(y+1)^2\\
			i=(0,3)&\quad \to\quad \frac{\delta_{212}f(1,-1)}{0!2!}(x-1)^1(y+1)^2
			%TODO WHAT?
		\end{align*}
		Tehát az együtthatók:
		\begin{align*}
			\delta_1f(x,y)&=2xy+y^2-2y;\quad \delta_1f(1,-1)=1\\
			\delta_2f(x,y)&=x^2+2xy-2y;\quad \delta_2f(1,-1)=-3\\
			\delta_{11}f(x,y)&=2y;\quad \delta_{11}f(1,-1)=-2\\
			\delta_{12}f(x,y)&=\delta_1(\delta_2f)(x,y)=2x+2y-2;\quad \delta_{12}f(1,-1)=-2\\
			\delta_{22}f(x,y)&=2x;\quad \delta_{22}f(1,-1)=2\\
			\delta_{111}f(x,y)&=\delta_1(\delta_{11}f)(x,y)=0\\
			\delta_{112}f(x,y)&=\delta_1(\delta_{12}f)(x,y)=2\\
			\delta_{122}f(x,y)&=\delta_1(\delta_{22}f)(x,y)=2\\
			\delta_{222}f(x,y)&=\delta_2(\delta_{22}f)(x,y)=0\\
		\end{align*}
		Ebből következik, hogy az összes $k\geq4$ esetén a $k$-adrendű parciális deriváltak mindegyike 0. Ezért, a negyedrendű hibatag 0, így a harmadrendű Taylor-polinom pontosan előállítja $f$-et.
		
		Tehát:
		\[ f(x,y)=T_3f(x,y)+0=f(1,-1)+\underbrace{\delta_1f(1,-1)(x-1)+\delta_2f(1,-1)(y+1)}_{k=1}+\]
		\[+\underbrace{\frac{1}{2}\delta_{11}f(1,-1)(x-1)^2+\delta_{12}f(1,-1)(x-1)(y+1)+\frac{1}{2}\delta_{22}f(1,-1)(y+1)^2}_{k=2}+\]
		\[+\underbrace{\frac{1}{6}\delta_{111}f(1,-1)(x-1)^3+\frac{1}{2}\delta_{112}f(1,-1)(x-1)^2(y+1)+\frac{1}{2}\delta_{122}f(1,-1)(x-1)(y+1)^2+\frac{1}{6}\delta_{222}f(1,-1)(y+1)^3}_{k=3} \]
		Tehát:
		\[ f(x,y)=2+(x-1)-3(y+1)-(x-1)^2-2(x-1)(y+1)+(y+1)^2+(x-1)^2(y+1)+(x-1)(y+1)^2 \quad (\forall(x,y)\in\R^2)\]
	\end{example}
